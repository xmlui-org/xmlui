<!--
  Shared icon tile card for icon grid views.

  Props:
  - `tileDimensions`: tile vars object from getTileVars(), passed by the parent icon view
  - `iconName`: icon identifier
  - `iconFallback`: optional fallback icon identifier
  - `name`: display name text
  - `badgeIcon`: optional icon name rendered inline after the filename (all screen sizes)
  - `handleContextMenu(ev)`: callable prop — right-click handler; if provided, enables right-click
  - `handleDotMenu(ev)`: callable prop — three-dot button handler; falls back to handleContextMenu if not provided
  - `secondaryText`: secondary row text value (optional)
-->
<Component name="IconTile">
  <Card when="{$props.interactive !== false}"
    width="{$props.width ?? '100%'}"
    height="{$props.tileDimensions.cardHeight}"
    horizontalAlignment="center"
    cursor="pointer"
    onClick="{emitEvent('tileClick')}"
    border="none"
    borderRadius="{$props.tileDimensions.borderRadius}"
    boxShadow="none"
    paddingTop="0"
    paddingRight="0"
    paddingBottom="{$props.tileDimensions.cardPadding}"
    paddingLeft="{$props.tileDimensions.cardPadding}"
    overflow="hidden"
    onContextMenu="ev => { if ($props.handleContextMenu) $props.handleContextMenu(ev); }"
    gap="0"
  >
    <!-- Three-dot button overlay: no top/right padding → button reaches physical card corner -->
    <HStack
      when="{!!$props.handleContextMenu || !!$props.handleDotMenu}"
      height="0px"
      width="100%"
      horizontalAlignment="end"
    >
      <Button
        variant="ghost"
        width="$space-6"
        height="$space-6"
        paddingHorizontal="0"
        marginTop="$space-7"
        marginRight="$space-1"
        zIndex="1"
        onClick="ev => {
          ev.stopPropagation();
          const handler = $props.handleDotMenu || $props.handleContextMenu;
          if (handler) handler(ev);
        }"
      >
        <Icon size="24px" name="btn-dotmenu" />
      </Button>
    </HStack>
    <Icon
      name="{$props.iconName}"
      size="{$props.tileDimensions.iconSize}"
      fallback="{$props.iconFallback ?? 'unknownfile'}"
      marginTop="{$props.tileDimensions.cardPadding}"
      marginLeft="{'-' + $props.tileDimensions.cardPadding }"
      marginBottom="{$props.tileDimensions.iconMarginBottom}"
    />
    <VStack gap="0" padding="0" paddingRight="{$props.tileDimensions.cardPadding}" width="100%" verticalAlignment="start">
      <!-- Name row: filename + optional badge icon inline -->
      <HStack
        gap="$space-1"
        verticalAlignment="center"
        width="100%"
        height="{$props.tileDimensions.nameHeight}"
      >
        <Text
          fontSize="{$props.tileDimensions.nameFontSize}"
          lineHeight="{$props.tileDimensions.nameLineHeight}"
          value="{$props.name}"
          maxLines="{$props.tileDimensions.nameMaxLines}"
          width="1*"
        />
        <Icon
          when="{!!$props.badgeIcon}"
          name="{$props.badgeIcon}"
          size="12px"
        />
      </HStack>
      <Text
        when="{!!$props.secondaryText}"
        fontSize="{$props.tileDimensions.secondaryTextFontSize}"
        color="$color-text-tertiary"
        value="{$props.secondaryText}"
        maxLines="1"
      />
    </VStack>
  </Card>
</Component>
